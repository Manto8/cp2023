void draw_white_sun(gdImagePtr img, int center_x, int center_y, int sun_radius, int color) {
    // M_PI 大小定義於 math.h 標頭檔中，因為三角函數中採用徑度為角度單位
    // 因此定義將角度轉為徑度的轉換變數為deg，角度值乘上deg就可轉為徑度
    浮點數 = M_PI/180;
    //依十二道輻射的每一尖角的角度為15度，求其對應直角三角形的另一角度為75度
    //求出十二道輻射中任一菱形的小半徑，菱形菱形的另一個對應小圓的半徑大小
    float sr = sun_radius/tan(75*deg);
    int ax、ay、bx、by、dx、dy、ex、ey；
    gdPoint點[4]；
    /* 在塗上十二道輻射中的單一菱形區域之前，先以座標點畫線測試是否正確
    斧頭 = 中心_x;
    ay = center_y - sun_radius；
    bx = center_x - sun_radius*tan(15*deg);
    通過=中心_y；
    前 = 中心_x;
    ey = center_y + sun_radius;
    dx = center_x + sun_radius*tan(15*deg);
    dy = 中心_y；
    //AB
    gdImageLine(img, ax, ay, bx, by, 顏色);
    // 是
    gdImageLine(img, bx, by, ex, ey, 顏色);
    // ED
    gdImageLine(img, ex, ey, dx, dy, 顏色);
    // DA
    gdImageLine(img, dx, dy, ax, ay, 顏色);
    */
    斧頭 = 中心_x;
    ay = center_y - sun_radius；
    bx = center_x - sun_radius*tan(15*deg);
    通過=中心_y；
    前 = 中心_x;
    ey = center_y + sun_radius;
    dx = center_x + sun_radius*tan(15*deg);
    dy = 中心_y；
    // 確定單一菱形區域的塗色正確後，利用迴圈每次轉動30度，總共轉六次即可塗上十二道光芒區域
    for (int i=1;i<=6;i++){
    // A
    點[0].x = ax+sun_radius*sin(30*deg*i);
    點[0].y = ay+sun_radius-sun_radius*cos(30*deg*i);
    // B
    點[1].x = bx+sr-sr*cos(30*deg*i);
    點[1].y = by-sr*sin(30*deg*i);
    // E
    點[2].x = ex-sun_radius*sin(30*deg*i);
    點[2].y = ey-(sun_radius-sun_radius*cos(30*deg*i));
    //D
    點[3].x = dx-(sr-sr*cos(30*deg*i));
    點[3].y = dy+sr*sin(30*deg*i);
    // 對菱形區域範圍塗色
    gdImageFilledPolygon(img, 點, 4, 顏色);
    // 在菱形區域外圍畫線，明確表示菱形範圍
    gdImagePolygon(img, 點, 4, 顏色);
    }
}
